二级跟随变化

siblings()同级单位
.eq()   获取第N个元素
代码:
	$(function  () {
    		$(".tab li").mouseenter(function  () {
    			$(this).addClass("active").siblings().removeClass("active")
    			var st=$(this).index()
    			$(".main").eq(st).addClass("selected").siblings().removeClass("selected")
    		})
    	})

轮播

代码:
$(function  () {
			var idx=1
			$(".wrap ol li").mouseenter(function  () {
				
				idx += 1;
				$(this).addClass("active").siblings().removeClass("active")
				$(".wrap ul li").eq($(this).index()).css({
					"left":650,
					"z-index":idx
				}).animate({
					"left":0	
				},300)
			})
		})

注意 css({})中间加花括号,属性需要加""
animate() 也一样
